/**
 * Дан массив целых чисел х длинной N
 * Массив упорядочен
 * Написать функцию которая из этого массива получит массив квадратов чисел
 * упорядоченный по возрастанию
 * [-3, 2, 4] -> [4, 9, 16]
 */

 /*
    1. Создадим массив результат длинной как входной массив
    2. Создадим два указателя
        первый в начале
        второй в конце
    3. Запустим цикл while
        пока l<=r
        каждую итерацию будем сравнивать
            правый с левым по их модулю - так при возведении в квадрат минус не учитывается
        если левый больше или равен правому
            добавляем квадрат левого в конец массива результата
            уменьшаем указатель в результате
            увеличиваем левый указатель на 1
        иначе
        добавляем квадрат правого в результат
            уменьшаем указатель результата на 1
            уменьшаем правый на 1

    Сложность:
        по времени O(n)
        по памяти O(2n)

    Примеры:
    []
    [-6 -6 -3 -1] -> [1 9 36 36]
 */
 public int[] getSqrs(int[] arr) {
    int[] result = new int[arr.length];

    int l = 0;
    int r = arr.length - 1;
    int i = result.length - 1;

    while (l <= r) {
        int left = Math.abs(arr[l]);
        int right = Math.abs(arr[r]);

        if (left >= right) {
            result[i] = left * left;
            i--;
            l++;
        } else {
            result[i] = right * right;
            i--;
            r--;
        }
    }

    return result;
 }
